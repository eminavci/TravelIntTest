def computeTopAirportsDf():
    chunks = pd.read_csv(bFile, sep="^", chunksize=1000000, usecols=[12, 34])
    bookinsPd = [chk.groupby('arr_port', sort=False)['pax'].sum().reset_index() for chk in chunks]
    return pd.concat(bookinsPd).groupby("arr_port").agg({'pax':'sum'}).reset_index().sort_values('pax', ascending=False)

from GeoBases import GeoBase

def getTopAirportsDf(n=10):
    topAirportsDetail = topAirports.iloc[:n]
    topAirportsDetail = topAirportsDetail.assign(city_name = None, airport_name = None)

    geo_o = GeoBase(data='ori_por', verbose=False)
    for index, row in topAirportsDetail.iterrows(): 
        geoMap = geo_o.get(row['arr_port'].strip(), None)
        topAirportsDetail.loc[index, 'arr_port'] = row['arr_port'].strip()
        topAirportsDetail.loc[index, 'city_name'] = geoMap['city_name_utf']
        topAirportsDetail.loc[index, 'airport_name'] = geoMap['name']
    return topAirportsDetail

import matplotlib.pyplot as plt
def obtainMonthlySearchPlt(dff):
    fig, ax = plt.subplots()
    labels = []

    for key, grp in dff.groupby(['Destination']):
        ax = grp.plot(ax=ax, kind='line', x='Date', y='Montly_search')
        labels.append(key)
    lines, _ = ax.get_legend_handles_labels()
    ax.legend(lines, labels, loc='best')
    plt.title('# of Airport Searches Monthly')
    return plt

import json
import cherrypy as appserver

class AirportsController():
    @appserver.expose
    @appserver.tools.json_out()
    def getTopAirports(self, n=10):
        data = {}
        data['status'] = 'success'
        data['key'] = 'value'
        data['n'] = int(n)
        data['topairports'] = json.loads(getTopAirportsDf(int(n)).reset_index().to_json(orient='records'))
        json_data = json.dumps(data)
 
        return json.loads(json_data)
    
    def showOnErr(status, message, traceback, version):
        data = {}
        data['status'] = 'error'
        data['status_code'] = status
        data['err_msg'] = message
        data['msg'] = "go to http://127.0.0.1:9000/getTopAirports?n=10"
        return json.dumps(data)

def startServer():
    appserver.tree.mount(AirportsController(), '/')
    appserver.config.update({'error_page.404': AirportsController.showOnErr})
    appserver.config.update({'error_page.500': AirportsController.showOnErr})
    appserver.config.update({'server.socket_port': 9000})
    appserver.engine.start()
startServer()
